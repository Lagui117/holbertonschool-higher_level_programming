#!/usr/bin/python3
# 1-main.py – tests pour 1-my_list.py

MyList = __import__('1-my_list').MyList

# --- instantiation / inherits from list / __str__ / append ---
my_list = MyList()
my_list.append(1)
my_list.append(4)
my_list.append(2)
my_list.append(3)
my_list.append(5)
print(my_list)              # [1, 4, 2, 3, 5]

# --- print_sorted() with sorted append ---
sorted_case = MyList([1, 2, 3])
sorted_case.append(4)
sorted_case.print_sorted()  # [1, 2, 3, 4]
print(sorted_case)          # [1, 2, 3, 4] (inchangé)

# --- print_sorted() with not sorted append ---
not_sorted = MyList([1, 3, 2])
not_sorted.append(0)
not_sorted.print_sorted()   # [0, 1, 2, 3]
print(not_sorted)           # [1, 3, 2, 0] (inchangé)

# --- print_sorted() with no sorted append with negative number ---
neg_case = MyList([1, 4, 2, 3, 5])
neg_case.append(-7)         # append d'un négatif dans une liste non triée
print(neg_case)             # [1, 4, 2, 3, 5, -7]
neg_case.print_sorted()     # [-7, 1, 2, 3, 4, 5]
print(neg_case)             # [1, 4, 2, 3, 5, -7] (inchangé)

# --- print_sorted() with empty list ---
empty = MyList()
empty.print_sorted()        # []

# --- print_sorted() returns a new list (i.e., does not mutate) ---
r = MyList([3, 1, 2])
before = str(r)
r.print_sorted()            # [1, 2, 3]
after = str(r)
print(before == after)      # True
